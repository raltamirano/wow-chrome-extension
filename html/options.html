Enter WOW mappings in the text area below in JSON format like: 
  [
		{
	 		"label": "Search on Google",
			"targetUrlPatterns": ["*://*/*"],
			"contexts": ["link", "selection"],
			"wowTemplate": "https://www.google.com.ar/search?q=${SOURCE_TEXT}"
		},
		{
	 		"label": "Expand short url (goo.gl)",
			"targetUrlPatterns": ["http://goo.gl/*", "https://goo.gl/*"],
			"contexts": ["link"],
			"wowTemplate": "https://www.googleapis.com/urlshortener/v1/url?shortUrl=${SOURCE_TEXT}"
		}
  ]
